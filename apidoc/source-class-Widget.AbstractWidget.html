138: abstract class AbstractWidget
139: {
140:     /**
141:      * The default dependence map
142:      *
143:      * @var array
144:      */
145:     protected $deps = array();
146: 
147:     /**
148:      * The widget container object
149:      *
150:      * @var Widget
151:      */
152:     protected $widget;
153: 
154:     /**
155:      * Constructor
156:      *
157:      * @param array $options The property options
158:      * @throws \InvalidArgumentException When option "widget" is not an instance of "Widget\Widget"
159:      */
160:     public function __construct(array $options = array())
161:     {
162:         $this->setOption($options);
163: 
164:         if (!isset($this->widget)) {
165:             $this->widget = Widget::create();
166:         } elseif (!$this->widget instanceof Widget) {
167:             throw new \InvalidArgumentException(sprintf('Option "widget" of class "%s" should be an instance of "Widget\Widget"', get_class($this)));
168:         }
169:     }
170: 
171:     /**
172:      * Set option property value
173:      *
174:      * @param string|array $name
175:      * @param mixed $value
176:      * @return AbstractWidget
177:      */
178:     public function setOption($name, $value = null)
179:     {
180:         // Set options
181:         if (is_array($name)) {
182:             if (is_array($value)) {
183:                 $name += array_intersect_key(get_object_vars($this), array_flip($value));
184:             }
185:             foreach ($name as $k => $v) {
186:                 $this->setOption($k, $v);
187:             }
188:             return $this;
189:         }
190: 
191:         // Append option
192:         if (isset($name[0]) && '+' == $name[0]) {
193:             $name = substr($name, 1);
194:             $this->$name = (array)$this->$name + $value;
195:             return $this;
196:         }
197: 
198:         if (method_exists($this, $method = 'set' . $name)) {
199:             return $this->$method($value);
200:         } else {
201:             $this->$name = $value;
202:             return $this;
203:         }
204:     }
205: 
206:     /**
207:      * Returns option property value
208:      *
209:      * @param string $name The name of property
210:      * @return mixed
211:      */
212:     public function getOption($name = null)
213:     {
214:         // Returns all property options
215:         if (null === $name) {
216:             return get_object_vars($this);
217:         }
218: 
219:         if (method_exists($this, $method = 'get' . $name)) {
220:             return $this->$method();
221:         } else {
222:             return isset($this->$name) ? $this->$name : null;
223:         }
224:     }
225: 
226:     /**
227:      * Invoke the widget by the given name
228:      *
229:      * @param string $name The name of widget
230:      * @param array $args The arguments for widget's __invoke method
231:      * @return mixed
232:      */
233:     public function __call($name, $args)
234:     {
235:         return call_user_func_array($this->$name, $args);
236:     }
237: 
238:     /**
239:      * Get the widget object by the given name
240:      *
241:      * @param  string $name The name of widget
242:      * @return AbstractWidget
243:      */
244:     public function __get($name)
245:     {
246:         return $this->$name = $this->widget->get($name, array(), $this->deps);
247:     }
248: }
