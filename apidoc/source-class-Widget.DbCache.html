 12:  * A cache widget that stored data in databases
 13:  *
 14:  * @author  Twin Huang <twinhuang@qq.com>
 15:  */
 16: class DbCache extends BaseCache
 17: {
 18:     /**
 19:      * The cache table name
 20:      *
 21:      * @var string
 22:      */
 23:     protected $table = 'cache';
 24: 
 25:     /**
 26:      * Whether check if table is exists or not
 27:      *
 28:      * @var bool
 29:      */
 30:     protected $checkTable = true;
 31: 
 32:     /**
 33:      * The SQL to check if table exists
 34:      *
 35:      * @var array
 36:      */
 37:     protected $checkTableSqls = array(
 38:         'mysql'     => "SHOW TABLES LIKE '%s'",
 39:         'sqlite'    => "SELECT name FROM sqlite_master WHERE type='table' AND name='%s'",
 40:         'pgsql'     => "SELECT true FROM pg_tables WHERE tablename = '%s'"
 41:     );
 42: 
 43:     /**
 44:      * The SQL to create cache table
 45:      *
 46:      * @var array
 47:      */
 48:     protected $createTableSqls = array(
 49:         'mysql'  => "CREATE TABLE %s (id VARCHAR(255) NOT NULL, value LONGTEXT NOT NULL, expire DATETIME NOT NULL, lastModified DATETIME NOT NULL, PRIMARY KEY(id)) DEFAULT CHARACTER SET utf8 COLLATE utf8_unicode_ci ENGINE = InnoDB",
 50:         'sqlite' => "CREATE TABLE %s (id VARCHAR(255) NOT NULL, value CLOB NOT NULL, expire DATETIME NOT NULL, lastModified DATETIME NOT NULL, PRIMARY KEY(id))",
 51:         'pgsql'  => "CREATE TABLE %s (id VARCHAR(255) NOT NULL, value TEXT NOT NULL, expire TIMESTAMP(0) WITHOUT TIME ZONE NOT NULL, lastModified TIMESTAMP(0) WITHOUT TIME ZONE NOT NULL, PRIMARY KEY(id))",
 52:     );
 53: 
 54:     /**
 55:      * Constructor
 56:      *
 57:      * @param array $options
 58:      */
 59:     public function __construct(array $options = array())
 60:     {
 61:         parent::__construct($options);
 62: 
 63:         $this->prepareTable();
 64:     }
 65: 
 66:     /**
 67:      * Check if table exists, if not exists, create table
 68:      */
 69:     public function prepareTable()
 70:     {
 71:         $db = $this->db;
 72:         $driver = $db->getDriver();
 73: 
 74:         $tableExistsSql = sprintf($this->checkTableSqls[$driver], $this->table);
 75:         if ($this->checkTable && !$this->db->fetchColumn($tableExistsSql)) {
 76:             $createTableSql = sprintf($this->createTableSqls[$driver], $this->table);
 77:             $db->executeUpdate($createTableSql);
 78:         }
 79:     }

 81:     /**
 82:      * {@inheritdoc}
 83:      */
 84:     protected function doGet($key)
 85:     {
 86:         $result = $this->db->select($this->table, $key);
 87:         return $result ? unserialize($result['value']) : false;
 88:     }
 89: 
 90:     /**
 91:      * {@inheritdoc}
 92:      */
 93:     protected function doSet($key, $value, $expire = 0)
 94:     {
 95:         $data = array(
 96:             'value' => serialize($value),
 97:             'lastModified' => date('Y-m-d H:i:s'),
 98:             'expire' => date('Y-m-d H:i:s', $expire ? time() + $expire: 2147483647)
 99:         );
100:         $identifier = array(
101:             'id' => $key
102:         );
103: 
104:         if ($this->doExists($key)) {
105:             // In mysql, the rowCount method return 0 when data is not modified,
106:             // so check errorCode to make sure it executed success
107:             $result = $this->db->update($this->table, $data, $identifier) || '0000' == $this->db->errorCode();
108:         } else {
109:             $result = $this->db->insert($this->table, $data + $identifier);
110:         }
111: 
112:         return (bool)$result;
113:     }
114:

116:      * {@inheritdoc}
117:      */
118:     protected function doRemove($key)

121:     }

124:      * {@inheritdoc}
125:      */
126:     protected function doExists($key)
127:     {
128:         return (bool)$this->db->select($this->table, $key);
129:     }

131:     /**
132:      * {@inheritdoc}
133:      */
134:     protected function doAdd($key, $value, $expire = 0)
135:     {
136:         if ($this->doExists($key)) {
137:             return false;
138:         } else {
139:             return $this->doSet($key, $value, $expire);
140:         }
141:     }
142: 
143:     /**
144:      * {@inheritdoc}
145:      */
146:     protected function doReplace($key, $value, $expire = 0)
147:     {
148:         if (!$this->doExists($key)) {
149:             return false;
150:         } else {
151:             return $this->doSet($key, $value, $expire);
152:         }
153:     }

156:      * Note: This method is not an atomic operation

159:      */
160:     protected function doIncr($key, $offset = 1)
161:     {
