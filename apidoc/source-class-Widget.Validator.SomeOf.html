 85:     {

 88:         $this->addError('atLeast');

 90:         $passed = 0;
 91:         $validator = null;
 92:         $props = array('name' => $this->name);
 93:         foreach ($this->rules as $rule => $options) {
 94:             if ($this->is->validateOne($rule, $input, $options, $validator, $props)) {
 95:                 $passed++;
 96:                 if ($passed >= $this->atLeast) {
 97:                     // Removes all error messages
 98:                     $this->errors = array();

100:                 }

102:                 $this->validators[$rule] = $validator;
103:             }
104:         }
105:

107:         $this->left = $this->atLeast - $passed;
108: 
109:         return false;

148:                 $messages[$rule] = implode(';', $validator->getMessages($name));

157:                 foreach ($validator->getMessages($name) as $option => $message) {

159:                 }
160:             }

162:         }
